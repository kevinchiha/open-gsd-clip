import { createChildLogger } from '../shared/logger.js';
import type { HostServices, AgentRole } from './types.js';
import { buildIssueTitle, buildIssueDescription, type AgentContext } from './context.js';

const log = createChildLogger('agent-invoker');

/**
 * Result of spawning an agent.
 */
export interface SpawnResult {
  issueId: string;
  runId: string;
}

/**
 * Spawn an agent by creating an issue and invoking the agent.
 *
 * @param services - HostServices for Paperclip API access
 * @param companyId - Company/tenant ID
 * @param agentId - Paperclip agent ID to invoke
 * @param ctx - Agent context with role, project path, phase, etc.
 * @returns SpawnResult with issue and run IDs
 */
export async function spawnAgent(
  services: HostServices,
  companyId: string,
  agentId: string,
  ctx: AgentContext,
): Promise<SpawnResult> {
  const title = buildIssueTitle(ctx);
  const description = buildIssueDescription(ctx);

  log.info({ role: ctx.role, agentId, title }, 'Creating issue for agent');

  const issue = await services.issues.create({
    companyId,
    title,
    description,
    status: 'todo',
    priority: 'high',
    assigneeAgentId: agentId,
    executionWorkspaceSettings: {
      mode: 'isolated',
    },
  });

  log.info({ role: ctx.role, issueId: issue.id }, 'Issue created, invoking agent');

  const { runId } = await services.agents.invoke({
    companyId,
    agentId,
    reason: `GSD ${ctx.role} task: ${title}`,
    prompt: `You have a new task assigned. Issue ID: ${issue.id}. Check your assigned issues and complete the task.`,
  });

  log.info({ role: ctx.role, issueId: issue.id, runId }, 'Agent invoked');

  return { issueId: issue.id, runId };
}
```

import type { GsdSignal } from '../signals/types.js';
import type { PhaseEvent, ErrorType } from '../pipeline/types.js';

/**
 * Map a GSD signal to a PhaseEvent for FSM dispatch.
 * Returns null for signals that don't map to phase events.
 */
export function mapSignalToPhaseEvent(signal: GsdSignal): PhaseEvent | null {
  switch (signal.type) {
    case 'PROJECT_READY':
      // PROJECT_READY is pipeline-level, not phase-level
      // Phase 4 will handle this specially
      return { type: 'STEP_COMPLETED' };

    case 'DISCUSS_COMPLETE':
      if (signal.status === 'success') {
        return { type: 'STEP_COMPLETED' };
      }
      return {
        type: 'STEP_FAILED',
        errorType: 'fatal' as ErrorType,
        message: signal.summary ?? 'Discussion failed',
      };

    case 'APPROVED':
      return { type: 'APPROVED' };

    case 'REVISION_NEEDED':
      return { type: 'REVISION_NEEDED' };

    case 'PLAN_COMPLETE':
      if (signal.status === 'success') {
        return { type: 'STEP_COMPLETED' };
      }
      return {
        type: 'STEP_FAILED',
        errorType: 'fatal' as ErrorType,
        message: signal.summary ?? 'Planning failed',
      };

    case 'EXECUTE_COMPLETE':
      if (signal.status === 'success') {
        return { type: 'STEP_COMPLETED' };
      }
      return {
        type: 'STEP_FAILED',
        errorType: 'fatal' as ErrorType,
        message: signal.summary ?? 'Execution failed',
      };

    case 'VERIFY_COMPLETE':
      return { type: 'STEP_COMPLETED' };

    case 'VERIFY_FAILED':
      return {
        type: 'STEP_FAILED',
        errorType: 'test_failure' as ErrorType,
        message: signal.issues.join('; '),
      };

    case 'AGENT_ERROR':
      return {
        type: 'STEP_FAILED',
        errorType: 'fatal' as ErrorType,
        message: signal.error,
      };

    case 'DECISION_NEEDED':
    case 'DECISION_MADE':
    case 'STALE_HEARTBEAT':
      // These signals don't directly map to phase events
      // They're handled by Phase 4 orchestration or monitoring
      return null;

    default:
      return null;
  }
}
```

```typescript
interface RunStatusEvent {
  type: 'heartbeat.run.status';
  data: {
    status: 'succeeded' | 'failed' | 'running' | 'pending';
    agentId: string;
    runId: string;
    issueId?: string;
  };
}

async function onEvent(params: unknown, id: string | number | null): Promise<JsonRpcResponse> {
  const event = params as { type: string; data: unknown };
  
  log.debug({ eventType: event.type }, 'Received event');

  // Handle run status events for agent completion
  if (event.type === 'heartbeat.run.status') {
    const run = event.data as RunStatusEvent['data'];
    
    // Only process terminal states
    if (run.status !== 'succeeded' && run.status !== 'failed') {
      return success({ received: true, status: run.status }, id);
    }

    log.info({
      status: run.status,
      agentId: run.agentId,
      runId: run.runId,
      issueId: run.issueId,
    }, 'Agent run completed');

    // If we have an issue ID, fetch comments and look for GSD_SIGNAL
    // Note: In Phase 4, we'll look up which phase has activeAgentIssueId matching run.issueId
    // For now, just log the completion
    
    if (run.issueId) {
      // Phase 4 will add: fetch comments via services.issues.listComments
      // For now, we just acknowledge the event
      log.info({ issueId: run.issueId }, 'Agent completed for issue (signal parsing in Phase 4)');
    }

    return success({
      received: true,
      status: run.status,
      // Phase 4 will add: signal, phaseEvent
    }, id);
  }

  // Default: just acknowledge
  return success({ received: true }, id);
}
```
