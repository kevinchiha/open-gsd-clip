import type { AgentRole } from './types.js';

/**
 * Context for building an agent's task issue.
 */
export interface AgentContext {
  role: AgentRole;
  projectPath: string;
  phaseNumber?: number;
  brief?: string;
  gsdCommand: string;
}

/**
 * Human-readable role names for issue titles.
 */
const ROLE_LABELS: Record<AgentRole, string> = {
  ceo: 'CEO',
  discusser: 'Discusser',
  planner: 'Planner',
  executor: 'Executor',
  verifier: 'Verifier',
};

/**
 * Build the issue title for an agent task.
 */
export function buildIssueTitle(ctx: AgentContext): string {
  const label = ROLE_LABELS[ctx.role];
  
  if (ctx.role === 'ceo') {
    return `${label}: Initialize project with ${ctx.gsdCommand}`;
  }
  
  const phase = ctx.phaseNumber ?? 0;
  return `${label}: Run ${ctx.gsdCommand} ${phase}`;
}
```

/**
 * GSD signal types for each role's completion signal.
 */
const ROLE_SIGNALS: Record<AgentRole, string> = {
  ceo: 'PROJECT_READY',
  discusser: 'DISCUSS_COMPLETE',
  planner: 'PLAN_COMPLETE',
  executor: 'EXECUTE_COMPLETE',
  verifier: 'VERIFY_COMPLETE',
};

/**
 * Build the issue description for an agent task.
 */
export function buildIssueDescription(ctx: AgentContext): string {
  const lines: string[] = [
    '## GSD Task',
    '',
    `**Project path:** ${ctx.projectPath}`,
    `**Command:** \`${ctx.gsdCommand}\``,
  ];

  // Add phase number for non-CEO roles
  if (ctx.role !== 'ceo' && ctx.phaseNumber !== undefined) {
    lines.push('', `**Phase:** ${ctx.phaseNumber}`);
  }

  // Add brief for CEO
  if (ctx.role === 'ceo' && ctx.brief) {
    lines.push('', '## Project Brief', '', ctx.brief);
    lines.push(
      '',
      '## Instructions',
      '',
      '1. Write the project brief above to a temp file, e.g., `/tmp/gsd-brief.md`',
      '2. Run: `claude /gsd:new-project --auto @/tmp/gsd-brief.md`',
      '3. After completion, post a GSD_SIGNAL comment on this issue',
    );
  }

  // Add "When Complete" section with signal template
  const signalType = ROLE_SIGNALS[ctx.role];
  const phaseForSignal = ctx.role === 'ceo' ? 0 : (ctx.phaseNumber ?? 0);

  lines.push(
    '',
    '## When Complete',
    '',
    'Post a comment on this issue with:',
    '',
    '```yaml',
    '---',
    `GSD_SIGNAL:${signalType}`,
    `phase: ${phaseForSignal}`,
    'status: success',
    'summary: [Brief description of what was done]',
    '---',
    '```',
  );

  return lines.join('\n');
}
```
