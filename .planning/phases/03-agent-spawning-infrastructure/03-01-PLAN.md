/**
 * GSD agent roles. Each role maps to a Paperclip agent definition
 * with specific instructions and GSD commands to run.
 */
export type AgentRole = 'ceo' | 'discusser' | 'planner' | 'executor' | 'verifier';

/**
 * All GSD roles as an array for iteration.
 */
export const AGENT_ROLES: AgentRole[] = ['ceo', 'discusser', 'planner', 'executor', 'verifier'];

/**
 * Maps GSD roles to Paperclip agent role strings.
 */
export const PAPERCLIP_ROLE_MAP: Record<AgentRole, string> = {
  ceo: 'ceo',
  discusser: 'engineer',
  planner: 'pm',
  executor: 'engineer',
  verifier: 'qa',
};

/**
 * Agent definition returned by the factory.
 */
export interface AgentDefinition {
  agentId: string;
  role: AgentRole;
  name: string;
}

/**
 * Configuration for the claude_local adapter.
 */
export interface AgentConfig {
  cwd: string;
  instructionsFilePath: string;
  model: string;
}

/**
 * Partial HostServices interface for GSD agent operations.
 * The full interface is provided by Paperclip at runtime.
 */
export interface HostServices {
  agents: {
    invoke(params: {
      companyId: string;
      agentId: string;
      reason?: string | null;
      prompt?: string;
    }): Promise<{ runId: string }>;
  };
  issues: {
    create(params: {
      companyId: string;
      title: string;
      description: string;
      status: string;
      priority: string;
      assigneeAgentId: string;
      executionWorkspaceSettings?: { mode: string };
    }): Promise<{ id: string; [key: string]: unknown }>;
    createComment(params: {
      companyId: string;
      issueId: string;
      body: string;
    }): Promise<{ id: string; [key: string]: unknown }>;
    listComments(params: {
      companyId: string;
      issueId: string;
    }): Promise<Array<{ id: string; body: string; [key: string]: unknown }>>;
  };
}

**Key implementation details:**

1. **Instructions directory:** Use `node:os.homedir()` + `.open-gsd-clip/agents` for stable path. Create directory on first call with `node:fs/promises.mkdir(..., { recursive: true })`.

2. **Instruction file content:** Read from embedded constants (not external files) for simplicity. Each role's instruction is a minimal template.

3. **Create-or-lookup:** The factory should attempt to list existing agents or check by name. Since HostServices may not have a `list` method, use a state-based approach: store created agent IDs in plugin state (HostServices.state if available) or use deterministic naming and attempt to use known names.

**Simplified approach for MVP:**
- Assume agents are created once and stored
- Use HostServices.issues.create with assigneeAgentId set to a known/created agent ID
- For agent creation: check if `agents.create` exists in HostServices. If not, log a warning and use a placeholder (agent must be created manually or via REST API)

import { mkdir, writeFile } from 'node:fs/promises';
import { join } from 'node:path';
import { createChildLogger } from '../shared/logger.js';

export function getInstructionsDir(): string {
  return join(homedir(), '.open-gsd-clip', 'agents');
}

export async function writeInstructionFile(role: AgentRole): Promise<string> {
  const dir = getInstructionsDir();
  await mkdir(dir, { recursive: true });
  const filePath = join(dir, `${role}.md`);
  await writeFile(filePath, INSTRUCTION_CONTENT[role], 'utf-8');
  log.debug({ role, path: filePath }, 'Wrote instruction file');
  return filePath;
}

export async function ensureAgentsExist(
  services: HostServices,
  companyId: string,
  projectPath: string,
  model: string,
): Promise<Record<AgentRole, AgentDefinition>> {
  const result = {} as Record<AgentRole, AgentDefinition>;

  for (const role of AGENT_ROLES) {
    const agentName = `gsd-${role}`;
    const instructionsPath = await writeInstructionFile(role);

    // Check if agents.create exists (may not be in HostServices)
    const createAgent = (services.agents as Record<string, unknown>).create as
      | ((params: Record<string, unknown>) => Promise<{ id: string }>)
      | undefined;

    if (createAgent) {
      const agent = await createAgent({
        companyId,
        name: agentName,
        role: PAPERCLIP_ROLE_MAP[role],
        title: `GSD ${role.charAt(0).toUpperCase() + role.slice(1)}`,
        adapterType: 'claude_local',
        adapterConfig: {
          cwd: projectPath,
          instructionsFilePath: instructionsPath,
          model,
        },
        permissions: { canCreateAgents: false },
      });
      result[role] = { agentId: agent.id, role, name: agentName };
      log.info({ role, agentId: agent.id }, 'Created agent');
    } else {
      // Fallback: return placeholder agent ID
      // In production, this would use REST API or require pre-created agents
      log.warn({ role }, 'agents.create not available -- using placeholder agent ID');
      result[role] = {
        agentId: `placeholder-${role}-${companyId}`,
        role,
        name: agentName,
      };
    }
  }

  return result;
}
